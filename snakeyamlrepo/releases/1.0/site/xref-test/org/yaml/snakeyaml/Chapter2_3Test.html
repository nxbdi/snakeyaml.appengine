32      public void testExample_2_15() {
33          String etalon = "Sammy Sosa completed another fine season with great stats.\n\n  63 Home Runs\n  0.288 Batting Average\n\nWhat a year!\n";
34          InputStream input = YamlDocument.class.getClassLoader().getResourceAsStream(
35                  YamlDocument.ROOT + "example2_15.yaml");
36          Yaml yaml = new Yaml();
37          String data = (String) yaml.load(input);
38          assertEquals(etalon, data);
39          //
40          String dumped = yaml.dump(data);
41          assertTrue(dumped.contains("Sammy Sosa completed another fine season with great stats"));
42          assertEquals("Must be splitted into 2 lines.", 2, dumped.split("\n").length);
43      }
