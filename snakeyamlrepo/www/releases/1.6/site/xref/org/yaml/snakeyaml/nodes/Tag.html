27  import org.yaml.snakeyaml.error.YAMLException;

29  public final class Tag implements Comparable<Tag> {
30      public static final String PREFIX = "tag:yaml.org,2002:";
31      public static final Tag YAML = new Tag(PREFIX + "yaml");
32      public static final Tag VALUE = new Tag(PREFIX + "value");
33      public static final Tag MERGE = new Tag(PREFIX + "merge");
34      public static final Tag SET = new Tag(PREFIX + "set");
35      public static final Tag PAIRS = new Tag(PREFIX + "pairs");
36      public static final Tag OMAP = new Tag(PREFIX + "omap");
37      public static final Tag BINARY = new Tag(PREFIX + "binary");
38      public static final Tag INT = new Tag(PREFIX + "int");
39      public static final Tag FLOAT = new Tag(PREFIX + "float");
40      public static final Tag TIMESTAMP = new Tag(PREFIX + "timestamp");
41      public static final Tag BOOL = new Tag(PREFIX + "bool");
42      public static final Tag NULL = new Tag(PREFIX + "null");
43      public static final Tag STR = new Tag(PREFIX + "str");
44      public static final Tag SEQ = new Tag(PREFIX + "seq");
45      public static final Tag MAP = new Tag(PREFIX + "map");
46      public static final Map<Tag, Set<Class<?>>> COMPATIBILITY_MAP;
47      static {
48          COMPATIBILITY_MAP = new HashMap<Tag, Set<Class<?>>>();
49          Set<Class<?>> floatSet = new HashSet<Class<?>>();
50          floatSet.add(Double.class);
51          floatSet.add(Float.class);
52          floatSet.add(BigDecimal.class);
53          COMPATIBILITY_MAP.put(FLOAT, floatSet);
54          //
55          Set<Class<?>> intSet = new HashSet<Class<?>>();
56          intSet.add(Integer.class);
57          intSet.add(Long.class);
58          intSet.add(BigInteger.class);
59          COMPATIBILITY_MAP.put(INT, intSet);
60          //
61          Set<Class<?>> timestampSet = new HashSet<Class<?>>();
62          timestampSet.add(Date.class);
63          timestampSet.add(java.sql.Date.class);
64          timestampSet.add(Timestamp.class);
65          COMPATIBILITY_MAP.put(TIMESTAMP, timestampSet);
66      }
67  
68      private final String value;
69  
70      public Tag(String tag) {
71          if (tag == null) {
72              throw new NullPointerException("Tag must be provided.");
73          } else if (tag.length() == 0) {
74              throw new IllegalArgumentException("Tag must not be empty.");
75          } else if (tag.trim().length() != tag.length()) {
76              throw new IllegalArgumentException("Tag must not contain leading or trailing spaces.");
77          }
78          this.value = tag;
79      }
80  
81      public Tag(Class<? extends Object> clazz) {
82          if (clazz == null) {
83              throw new NullPointerException("Class for tag must be provided.");
84          }
85          this.value = Tag.PREFIX + clazz.getName();
86      }
87  
88      public String getValue() {
89          return value;
90      }
91  
92      public boolean startsWith(String prefix) {
93          return value.startsWith(prefix);
94      }
95  
96      public String getClassName() {
97          if (!value.startsWith(Tag.PREFIX)) {
98              throw new YAMLException("Unknown tag: " + value);
99          }
100         return value.substring(Tag.PREFIX.length());
101     }
102 
103     public int getLength() {
104         return value.length();
105     }
106 
107     @Override
108     public String toString() {
109         return value;
110     }
111 
112     @Override
113     public boolean equals(Object obj) {
114         if (obj == null) {
115             return false;
116         }
117         if (obj instanceof Tag) {
118             return value.equals(((Tag) obj).getValue());
119         } else if (obj instanceof String) {
120             if (value.equals(obj.toString())) {
121                 // TODO to be removed later (version 2.0?)
122                 System.err.println("Comparing Tag and String is deprecated.");
123                 return true;
124             }
125         }
126         return false;
127     }
128 
129     @Override
130     public int hashCode() {
131         return value.hashCode();
132     }
133 
134     /***
135      * Java has more then 1 class compatible with a language-independent tag
136      * (!!int, !!float, !!timestamp etc)
137      * 
138      * @param clazz
139      *            - Class to check compatibility
140      * @return true when the Class can be represented by this
141      *         language-independent tag
142      */
143     public boolean isCompatible(Class<?> clazz) {
144         Set<Class<?>> set = COMPATIBILITY_MAP.get(this);
145         if (set != null) {
146             return set.contains(clazz);
147         } else {
148             return false;
149         }
150     }
151 
152     /***
153      * Check whether this tag matches the global tag for the Class
154      * 
155      * @param clazz
156      *            - Class to check
157      * @return true when the this tag can be used as a global tag for the Class
158      */
159     public boolean matches(Class<? extends Object> clazz) {
160         return value.equals(Tag.PREFIX + clazz.getName());
161     }
162 
163     public int compareTo(Tag o) {
164         return value.compareTo(o.getValue());
165     }
