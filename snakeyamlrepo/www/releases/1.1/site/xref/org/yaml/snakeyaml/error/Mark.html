6   import org.yaml.snakeyaml.scanner.ScannerImpl;

8   /***
9    * It's just a record and its only use is producing nice error messages. Parser
10   * does not use it for any other purposes.
11   * 
12   * @see <a href="http://pyyaml.org/wiki/PyYAML">PyYAML</a> for more information
13   */
14  public final class Mark {
15      private String name;
16      private int line;
17      private int column;
18      private String buffer;
19      private int pointer;
20  
21      public Mark(String name, int index, int line, int column, String buffer, int pointer) {
22          super();
23          this.name = name;
24          this.line = line;
25          this.column = column;
26          this.buffer = buffer;
27          this.pointer = pointer;
28      }
29  
30      private boolean isLineBreak(char ch) {
31          return ScannerImpl.NULL_OR_LINEBR.indexOf(ch) != -1;
32      }
33  
34      public String get_snippet(int indent, int max_length) {
35          if (buffer == null) {
36              return null;
37          }
38          float half = max_length / 2 - 1;
39          int start = pointer;
40          String head = "";
41          while ((start > 0) && (!isLineBreak(buffer.charAt(start - 1)))) {
42              start -= 1;
43              if (pointer - start > half) {
44                  head = " ... ";
45                  start += 5;
46                  break;
47              }
48          }
49          String tail = "";
50          int end = pointer;
51          while ((end < buffer.length()) && (!isLineBreak(buffer.charAt(end)))) {
52              end += 1;
53              if (end - pointer > half) {
54                  tail = " ... ";
55                  end -= 5;
56                  break;
57              }
58          }
59          String snippet = buffer.substring(start, end);
60          StringBuffer result = new StringBuffer();
61          for (int i = 0; i < indent; i++) {
62              result.append(" ");
63          }
64          result.append(head);
65          result.append(snippet);
66          result.append(tail);
67          result.append("\n");
68          for (int i = 0; i < indent + pointer - start + head.length(); i++) {
69              result.append(" ");
70          }
71          result.append("^");
72          return result.toString();
73      }
74  
75      public String get_snippet() {
76          return get_snippet(4, 75);
77      }
78  
79      @Override
80      public String toString() {
81          String snippet = get_snippet();
82          StringBuffer where = new StringBuffer(" in \"");
83          where.append(name);
84          where.append("\", line ");
85          where.append(line);
86          where.append(", column ");
87          where.append(column);
88          if (snippet != null) {
89              where.append(":\n");
90              where.append(snippet);
91          }
92          return where.toString();
93      }
94  
95      public String getName() {
96          return name;
97      }

99      public int getLine() {
100         return line;
101     }

104         return column;
105     }
