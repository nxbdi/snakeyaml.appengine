4   package org.yaml.snakeyaml.introspector;
5   
6   import java.lang.reflect.Field;
7   
8   import org.yaml.snakeyaml.error.YAMLException;
9   
10  public class FieldProperty extends Property {
11      private final Field field;
12  
13      public FieldProperty(Field field) {
14          super(field.getName(), field.getType());
15          this.field = field;
16      }
17  
18      @Override
19      public void set(Object object, Object value) throws Exception {
20          field.set(object, value);
21      }
22  
23      @Override
24      public Object get(Object object) {
25          try {
26              return field.get(object);
27          } catch (Exception e) {
28              throw new YAMLException("Unable to access field " + field.getName() + " on object "
29                      + object + " : " + e);
30          }
31      }
32  }
