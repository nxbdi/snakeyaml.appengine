12  public abstract class Token {
13      private final Mark startMark;
14      private final Mark endMark;
15  
16      public Token(Mark startMark, Mark endMark) {
17          if (startMark == null || endMark == null) {
18              throw new YAMLException("Token requires marks.");
19          }
20          this.startMark = startMark;
21          this.endMark = endMark;
22      }
23  
24      public String toString() {
25          return "<" + this.getClass().getName() + "(" + getArguments() + ")>";
26      }
27  
28      public Mark getStartMark() {
29          return startMark;
30      }
31  
32      public Mark getEndMark() {
33          return endMark;
34      }
35  
36      /***
37       * @see __repr__ for Token in PyYAML
38       */
39      protected String getArguments() {
40          return "";
41      }
42  
43      /***
44       * For error reporting.
45       * 
46       * @see class variable 'id' in PyYAML
47       */
48      public abstract String getTokenId();
49  
50      /*
51       * for tests only
52       */
53      @Override
54      public boolean equals(Object obj) {
55          if (obj instanceof Token) {
56              return toString().equals(obj.toString());
57          } else {
58              return false;
59          }
60      }
