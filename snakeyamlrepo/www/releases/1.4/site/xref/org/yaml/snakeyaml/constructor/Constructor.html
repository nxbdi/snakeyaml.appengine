49  public class Constructor extends SafeConstructor {
50      private final Map<String, Class<? extends Object>> typeTags;
51      private final Map<Class<? extends Object>, TypeDescription> typeDefinitions;
52  
53      public Constructor() {
54          this(Object.class);
55      }
56  
57      public Constructor(Class<? extends Object> theRoot) {
58          if (theRoot == null) {
59              throw new NullPointerException("Root type must be provided.");
60          }
61          this.yamlConstructors.put(null, new ConstructYamlObject());
62          rootType = theRoot;
63          typeTags = new HashMap<String, Class<? extends Object>>();
64          typeDefinitions = new HashMap<Class<? extends Object>, TypeDescription>();
65          yamlClassConstructors.put(NodeId.scalar, new ConstructScalar());
66          yamlClassConstructors.put(NodeId.mapping, new ConstructMapping());
67          yamlClassConstructors.put(NodeId.sequence, new ConstructSequence());
68      }

70      /***
71       * Create Constructor for a class which does not have to be in the classpath
72       * or for a definition from a Spring ApplicationContext.
73       * 
74       * @param theRoot
75       *            fully qualified class name of the root JavaBean
76       * @throws ClassNotFoundException
77       */
78      public Constructor(String theRoot) throws ClassNotFoundException {
79          this(Class.forName(check(theRoot)));
80      }

82      private static final String check(String s) {
83          if (s == null) {
84              throw new NullPointerException("Root type must be provided.");
85          }
86          if (s.trim().length() == 0) {
87              throw new YAMLException("Root type must be provided.");
88          }
89          return s;
90      }

92      /***
93       * Make YAML aware how to parse a custom Class. If there is no root Class
94       * assigned in constructor then the 'root' property of this definition is
95       * respected.
96       * 
97       * @param definition
98       *            to be added to the Constructor
99       * @return the previous value associated with <tt>definition</tt>, or
100      *         <tt>null</tt> if there was no mapping for <tt>definition</tt>.
101      */
102     public TypeDescription addTypeDescription(TypeDescription definition) {
103         if (definition == null) {
104             throw new NullPointerException("TypeDescription is required.");
105         }
106         if (rootType == Object.class && definition.isRoot()) {
107             rootType = definition.getType();
108         }
109         String tag = definition.getTag();
110         typeTags.put(tag, definition.getType());
111         return typeDefinitions.put(definition.getType(), definition);
112     }

114     /***
115      * Construct mapping instance (Map, JavaBean) when the runtime class is
116      * known.
117      */
118     private class ConstructMapping implements Construct {
119         /***
120          * Construct JavaBean. If type safe collections are used please look at
121          * <code>TypeDescription</code>.
122          * 
123          * @param node
124          *            node where the keys are property names (they can only be
125          *            <code>String</code>s) and values are objects to be created
126          * @return constructed JavaBean
127          */
128         public Object construct(Node node) {
129             MappingNode mnode = (MappingNode) node;
130             if (Map.class.isAssignableFrom(node.getType())) {
131                 if (node.isTwoStepsConstruction()) {
132                     // TODO when the Map implementation is known it should be
133                     // used
134                     return createDefaultMap();
135                 } else {
136                     return constructMapping(mnode);
137                 }
138             } else if (Set.class.isAssignableFrom(node.getType())) {
139                 if (node.isTwoStepsConstruction()) {
140                     // TODO when the Set implementation is known it should be
141                     // used
142                     return createDefaultSet();
143                 } else {
144                     return constructSet(mnode);
145                 }
146             } else {
147                 if (node.isTwoStepsConstruction()) {
148                     return createEmptyJavaBean(mnode);
149                 } else {
150                     return constructJavaBean2ndStep(mnode, createEmptyJavaBean(mnode));
151                 }
152             }
153         }
154 
155         @SuppressWarnings("unchecked")
156         public void construct2ndStep(Node node, Object object) {
157             if (Map.class.isAssignableFrom(node.getType())) {
158                 constructMapping2ndStep((MappingNode) node, (Map<Object, Object>) object);
159             } else if (Set.class.isAssignableFrom(node.getType())) {
160                 constructSet2ndStep((MappingNode) node, (Set<Object>) object);
161             } else {
162                 constructJavaBean2ndStep((MappingNode) node, object);
163             }
164         }

166         private Object createEmptyJavaBean(MappingNode node) {
167             try {
168                 Class<? extends Object> type = node.getType();
169                 if (Modifier.isAbstract(type.getModifiers())) {
170                     node.setType(getClassForNode(node));
171                 }
172                 /***
173                  * Using only default constructor. Everything else will be
174                  * initialized on 2nd step. If we do here some partial
175                  * initialization, how do we then track what need to be done on
176                  * 2nd step? I think it is better to get only object here (to
177                  * have it as reference for recursion) and do all other thing on
178                  * 2nd step.
179                  */
180                 return node.getType().newInstance();
181             } catch (InstantiationException e) {
182                 throw new YAMLException(e);
183             } catch (IllegalAccessException e) {
184                 throw new YAMLException(e);
185             }
186         }

188         @SuppressWarnings("unchecked")
189         private Object constructJavaBean2ndStep(MappingNode node, Object object) {
190             Class<? extends Object> beanType = node.getType();
191             List<NodeTuple> nodeValue = (List<NodeTuple>) node.getValue();
192             for (NodeTuple tuple : nodeValue) {
193                 ScalarNode keyNode;
194                 if (tuple.getKeyNode() instanceof ScalarNode) {
195                     // key must be scalar
196                     keyNode = (ScalarNode) tuple.getKeyNode();
197                 } else {
198                     throw new YAMLException("Keys must be scalars but found: " + tuple.getKeyNode());
199                 }
200                 Node valueNode = tuple.getValueNode();
201                 // keys can only be Strings
202                 keyNode.setType(String.class);
203                 String key = (String) constructObject(keyNode);
204                 boolean isArray = false;
205                 try {
206                     Property property = getProperty(beanType, key);
207                     valueNode.setType(property.getType());
208                     TypeDescription memberDescription = typeDefinitions.get(beanType);
209                     if (memberDescription != null) {
210                         switch (valueNode.getNodeId()) {
211                         case sequence:
212                             SequenceNode snode = (SequenceNode) valueNode;
213                             Class<? extends Object> memberType = memberDescription
214                                     .getListPropertyType(key);
215                             if (memberType != null) {
216                                 snode.setListType(memberType);
217                             } else if (property.getType().isArray()) {
218                                 isArray = true;
219                                 snode.setListType(property.getType().getComponentType());
220                             }
221                             break;
222                         case mapping:
223                             MappingNode mnode = (MappingNode) valueNode;
224                             Class<? extends Object> keyType = memberDescription.getMapKeyType(key);
225                             if (keyType != null) {
226                                 mnode.setKeyType(keyType);
227                                 mnode.setValueType(memberDescription.getMapValueType(key));
228                             }
229                             break;
230                         }
231                     }
232                     Object value = constructObject(valueNode);
233                     if (isArray) {
234                         List<Object> list = (List<Object>) value;
235                         value = list.toArray(createArray(property.getType()));
236                     }
237                     property.set(object, value);
238                 } catch (Exception e) {
239                     throw new YAMLException("Cannot create property=" + key + " for JavaBean="
240                             + object + "; " + e.getMessage(), e);
241                 }
242             }
243             return object;
244         }

246         @SuppressWarnings("unchecked")
247         private <T> T[] createArray(Class<T> type) {
248             return (T[]) Array.newInstance(type.getComponentType(), 0);
249         }

251         private Property getProperty(Class<? extends Object> type, String name)
252                 throws IntrospectionException {
253             for (PropertyDescriptor property : Introspector.getBeanInfo(type)
254                     .getPropertyDescriptors()) {
255                 if (property.getName().equals(name)) {
256                     if (property.getWriteMethod() != null) {
257                         return new MethodProperty(property);
258                     } else {
259                         throw new YAMLException("Property '" + name + "' on JavaBean: "
260                                 + type.getName() + " does not have the write method");
261                     }
262                 }
263             }
264             for (Field field : type.getFields()) {
265                 int modifiers = field.getModifiers();
266                 if (Modifier.isStatic(modifiers) || Modifier.isTransient(modifiers)) {
267                     continue;
268                 }
269                 if (field.getName().equals(name)) {
270                     return new FieldProperty(field);
271                 }
272             }
273             throw new YAMLException("Unable to find property '" + name + "' on class: "
274                     + type.getName());
275         }

278     /***
279      * Construct an instance when the runtime class is not known but a global
280      * tag with a class name is defined. It delegates the construction to the
281      * appropriate constructor based on the node kind (scalar, sequence,
282      * mapping)
283      */
284     private class ConstructYamlObject extends AbstractConstruct {
285         @SuppressWarnings("unchecked")
286         public Object construct(Node node) {
287             Object result = null;
288             try {
289                 Class cl = getClassForNode(node);
290                 node.setType(cl);
291                 // call the constructor as if the runtime class is defined
292                 Construct constructor = yamlClassConstructors.get(node.getNodeId());
293                 result = constructor.construct(node);
294             } catch (Exception e) {
295                 throw new ConstructorException(null, null, "Can't construct a java object for "
296                         + node.getTag() + "; exception=" + e.getMessage(), node.getStartMark(), e);
297             }
298             return result;
299         }
300     }

302     /***
303      * Construct scalar instance when the runtime class is known. Recursive
304      * structures are not supported.
305      */
306     protected class ConstructScalar extends AbstractConstruct {
307         @SuppressWarnings("unchecked")
308         public Object construct(Node nnode) {
309             ScalarNode node = (ScalarNode) nnode;
310             Class type = node.getType();
311             Object result;
312             if (type.isPrimitive() || type == String.class || Number.class.isAssignableFrom(type)
313                     || type == Boolean.class || Date.class.isAssignableFrom(type)
314                     || type == Character.class || type == BigInteger.class
315                     || Enum.class.isAssignableFrom(type) || Tags.BINARY.equals(node.getTag())) {
316                 // standard classes created directly
317                 result = constructStandardJavaInstance(type, node);
318             } else {
319                 // there must be only 1 constructor with 1 argument
320                 java.lang.reflect.Constructor[] javaConstructors = type.getConstructors();
321                 int oneArgCount = 0;
322                 java.lang.reflect.Constructor javaConstructor = null;
323                 for (java.lang.reflect.Constructor c : javaConstructors) {
324                     if (c.getParameterTypes().length == 1) {
325                         oneArgCount++;
326                         javaConstructor = c;
327                     }
328                 }
329                 Object argument;
330                 if (javaConstructor == null) {
331                     throw new YAMLException("No single argument constructor found for " + type);
332                 } else if (oneArgCount == 1) {
333                     argument = constructStandardJavaInstance(
334                             javaConstructor.getParameterTypes()[0], node);
335                 } else {
336                     // TODO it should be possible to use implicit types instead
337                     // of forcing String. Resolver must be available here to
338                     // obtain the implicit tag. Then we can set the tag and call
339                     // callConstructor(node) to create the argument instance.
340                     // On the other hand it may be safer to require a custom
341                     // constructor to avoid guessing the argument class
342                     argument = constructScalar(node);
343                     try {
344                         javaConstructor = type.getConstructor(String.class);
345                     } catch (Exception e) {
346                         throw new ConstructorException(null, null,
347                                 "Can't construct a java object for scalar " + node.getTag()
348                                         + "; No String constructor found. Exception="
349                                         + e.getMessage(), node.getStartMark(), e);
350                     }
351                 }
352                 try {
353                     result = javaConstructor.newInstance(argument);
354                 } catch (Exception e) {
355                     throw new ConstructorException(null, null,
356                             "Can't construct a java object for scalar " + node.getTag()
357                                     + "; exception=" + e.getMessage(), node.getStartMark(), e);
358                 }
359             }
360             return result;
361         }

363         @SuppressWarnings("unchecked")
364         private Object constructStandardJavaInstance(Class type, ScalarNode node) {
365             Object result;
366             if (type == String.class) {
367                 Construct stringConstructor = yamlConstructors.get(Tags.STR);
368                 result = stringConstructor.construct((ScalarNode) node);
369             } else if (type == Boolean.class || type == Boolean.TYPE) {
370                 Construct boolConstructor = yamlConstructors.get(Tags.BOOL);
371                 result = boolConstructor.construct((ScalarNode) node);
372             } else if (type == Character.class || type == Character.TYPE) {
373                 Construct charConstructor = yamlConstructors.get(Tags.STR);
374                 String ch = (String) charConstructor.construct((ScalarNode) node);
375                 if (ch.length() == 0) {
376                     result = null;
377                 } else if (ch.length() != 1) {
378                     throw new YAMLException("Invalid node Character: '" + ch + "'; length: "
379                             + ch.length());
380                 } else {
381                     result = new Character(ch.charAt(0));
382                 }
383             } else if (Date.class.isAssignableFrom(type)) {
384                 Construct dateConstructor = yamlConstructors.get(Tags.TIMESTAMP);
385                 Date date = (Date) dateConstructor.construct((ScalarNode) node);
386                 if (type == Date.class) {
387                     result = date;
388                 } else {
389                     try {
390                         java.lang.reflect.Constructor<?> constr = type.getConstructor(long.class);
391                         result = constr.newInstance(date.getTime());
392                     } catch (Exception e) {
393                         throw new YAMLException("Cannot construct: '" + type + "'");
394                     }
395                 }
396             } else if (type == Float.class || type == Double.class || type == Float.TYPE
397                     || type == Double.TYPE || type == BigDecimal.class) {
398                 Construct doubleConstructor = yamlConstructors.get(Tags.FLOAT);
399                 result = doubleConstructor.construct(node);
400                 if (type == Float.class || type == Float.TYPE) {
401                     result = new Float((Double) result);
402                 } else if (type == BigDecimal.class) {
403                     result = new BigDecimal(((Double) result).doubleValue());
404                 }
405             } else if (type == Byte.class || type == Short.class || type == Integer.class
406                     || type == Long.class || type == BigInteger.class || type == Byte.TYPE
407                     || type == Short.TYPE || type == Integer.TYPE || type == Long.TYPE) {
408                 Construct intConstructor = yamlConstructors.get(Tags.INT);
409                 result = intConstructor.construct(node);
410                 if (type == Byte.class || type == Byte.TYPE) {
411                     result = new Byte(result.toString());
412                 } else if (type == Short.class || type == Short.TYPE) {
413                     result = new Short(result.toString());
414                 } else if (type == Integer.class || type == Integer.TYPE) {
415                     result = new Integer(result.toString());
416                 } else if (type == Long.class || type == Long.TYPE) {
417                     result = new Long(result.toString());
418                 } else {
419                     // only BigInteger left
420                     result = new BigInteger(result.toString());
421                 }
422             } else if (Enum.class.isAssignableFrom(type)) {
423                 String enumValueName = node.getValue();
424                 try {
425                     result = Enum.valueOf(type, enumValueName);
426                 } catch (Exception ex) {
427                     throw new YAMLException("Unable to find enum value '" + enumValueName
428                             + "' for enum class: " + type.getName());
429                 }
430             } else {
431                 throw new YAMLException("Unsupported class: " + type);
432             }
433             return result;
434         }

437     /***
438      * Construct sequence (List, Array, or immutable object) when the runtime
439      * class is known.
440      */
441     private class ConstructSequence implements Construct {
442         @SuppressWarnings("unchecked")
443         public Object construct(Node node) {
444             SequenceNode snode = (SequenceNode) node;
445             if (List.class.isAssignableFrom(node.getType()) || node.getType().isArray()) {
446                 if (node.isTwoStepsConstruction()) {
447                     return createDefaultList(snode.getValue().size());
448                 } else {
449                     return constructSequence(snode);
450                 }
451             } else {
452                 // create immutable object
453                 List<java.lang.reflect.Constructor> possibleConstructors = new LinkedList<java.lang.reflect.Constructor>();
454                 for (java.lang.reflect.Constructor constructor : node.getType().getConstructors()) {
455                     if (snode.getValue().size() == constructor.getParameterTypes().length) {
456                         possibleConstructors.add(constructor);
457                     }
458                 }
459                 if (possibleConstructors.isEmpty()) {
460                     throw new YAMLException("No constructors with "
461                             + String.valueOf(snode.getValue().size()) + " arguments found for "
462                             + node.getType());
463                 }
464                 List<Object> argumentList;
465                 if (possibleConstructors.size() == 1) {
466                     argumentList = new LinkedList<Object>();
467                     java.lang.reflect.Constructor c = possibleConstructors.get(0);
468                     int index = 0;
469                     for (Node argumentNode : snode.getValue()) {
470                         Class type = c.getParameterTypes()[index];
471                         // set runtime classes for arguments
472                         argumentNode.setType(type);
473                         Object argumentValue = constructObject(argumentNode);
474                         argumentList.add(argumentValue);
475                         index++;
476                     }
477                 } else {
478                     // use BaseConstructor
479                     argumentList = (List<Object>) constructSequence(snode);
480                 }
481                 Class[] parameterTypes = new Class[argumentList.size()];
482                 int index = 0;
483                 for (Object parameter : argumentList) {
484                     parameterTypes[index] = parameter.getClass();
485                     index++;
486                 }
487                 java.lang.reflect.Constructor javaConstructor;
488                 try {
489                     Class cl = node.getType();
490                     javaConstructor = cl.getConstructor(parameterTypes);
491                     Object[] initargs = argumentList.toArray();
492                     return javaConstructor.newInstance(initargs);
493                 } catch (Exception e) {
494                     throw new YAMLException(e);
495                 }
496             }
497         }

501             SequenceNode snode = (SequenceNode) node;

511     protected Class<?> getClassForNode(Node node) {
512         Class<? extends Object> classForTag = typeTags.get(node.getTag());
513         if (classForTag == null) {
514             if (node.getTag().length() < Tags.PREFIX.length()) {
515                 throw new YAMLException("Unknown tag: " + node.getTag());
516             }
517             String name = node.getTag().substring(Tags.PREFIX.length());
518             Class<?> cl;
519             try {
520                 cl = Class.forName(name);
521             } catch (ClassNotFoundException e) {
522                 throw new YAMLException("Class not found: " + name);
523             }
524             typeTags.put(node.getTag(), cl);
525             return cl;
526         } else {
527             return classForTag;
528         }
